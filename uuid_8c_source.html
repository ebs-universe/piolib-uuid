107 static void _uuid1(uuid_t * out, void (*ts_writer)(uuid_t *, uint8_t *)){
108  memcpy(out, &uuid1_template, sizeof(uuid_t));
109  // Write time
110  tm_system_t stime;
111  union TS{
112  uint64_t i;
113  uint8_t b[8];
114  }ts;
115  tm_current_time(&stime);
116  ts.i = stime.seconds * 10000000 + stime.frac * TIME_TICKS_PER_SECOND * 10 + UUID_EPOCH_OFFSET;
117  if (ts.i == uuid1_state.ts_last){
118  uuid1_state.uuids_this_tick++;
119  }
120  else{
121  uuid1_state.ts_last = ts.i;
122  uuid1_state.uuids_this_tick = 0;
123  }
124  ts.i += uuid1_state.uuids_this_tick;
125  ts_writer(out, &(ts.b[0]));
126 }

164 void uuid3(uuid_t * out, uuid_t * ns, uint8_t * name_p, uint8_t len){
165  md5_init(&md5ctx);
166  uint8_t t;
167 
168  memcpy((void *)&(hashstage[0]), ns, sizeof(uuid_t));
169  t = MD5_BLOCK_BYTES - sizeof(uuid_t);
170  if (len <= t){
171  t = len;
172  }
173  memcpy((void *)&(hashstage[sizeof(uuid_t)]), name_p, t);
174  name_p += t;
175  len -= t;
176  if (!len){
177  md5_lastBlock(&md5ctx, &(hashstage[0]), (sizeof(uuid_t) + t) * 8);
178  }
179  else{
180  md5_nextBlock(&md5ctx, &(hashstage[0]));
181  while (len){
182  if (len <= MD5_BLOCK_BYTES){
183  md5_lastBlock(&md5ctx, name_p, len * 8);
184  len = 0;
185  }
186  else{
187  md5_nextBlock(&md5ctx, name_p);
188  name_p += MD5_BLOCK_BYTES;
189  len -= MD5_BLOCK_BYTES;
190  }
191  }
192  }
193  md5_ctx2hash((md5_hash_t *)&(out->b), &md5ctx);
194  _uuid_setversion(out, (3<<4));
195 }
